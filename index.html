1st variant


	let distances = [
{ from: 'Minsk', to: 'Vitebsk', distance: 270},
{ from: 'Vitebsk', to: 'Homel', distance: 330},
{ from: 'Homel', to: 'Brest', distance: 533}
];

	if(distances[i].distance > 300){ 
	total +=distances[i].distance * 0.621371;
  }
}

console.log(Math.round(total)) // 536

    
			
2nd variant


let distances = [
	{ from: 'Minsk', to: 'Vitebsk', distance: 270},
	{ from: 'Vitebsk', to: 'Homel', distance: 330},
	{ from: 'Homel', to: 'Brest', distance: 533}
	];

let total = distances.filter(item => 
item.distance > 300) 
  .map(item=> item.distance * 0.621371)
  .reduce((prev, distance) => prev + distance, 0);



console.log(Math.round(total)) // 536

KISS in coding

Short-circuit Evaluation

							
Before

							
const variable;

if (variable === null || variable === undefined || 
variable === '') {
console.log("VARIABLE NOT FOUND!")
} else {
console.log(variable)
}
							
						
							
After

							
const variable;

console.log( variable || "VARIABLE NOT FOUND!" )

KISS in coding

Arrow functions

							
Before

							
function sayHello(name) {
console.log('Hello', name)
}
							
						
							
After

							
const sayHello = name => console.log("Hello", name)

note! Arrow Functions:

  1. Don't have “this”.
  2. Do Not have “arguments”.
  3. Cannot be called with “new”.
  4. Also don't have “super”.

KISS in coding

Array.forEach() method

							
Before

							
const list = [2, 5, 7, 2, 6, 2, 3, 5];
for (let i = 0; i < list.length; i++) { 
	console.log(list[i]) 
}
							
						
							
After

							
const list = [2, 5, 7, 2, 6, 2, 3, 5];
list.forEach(number => console.log(number))

KISS in coding

Default Parameters

							
Before

							
function volume(l, w, h) {
if (w === undefined)
w = 1;
if (h === undefined)
h = 1;
return l * w * h;
}
							
						
							
After

							
const volume = ( l, w = 1, h = 1) => l * w * h;

KISS in coding

Destructuring

							
Before

							
const person = {
first: 'Jan',
last: 'Chorny',
country: 'Belarus',
city: 'Vitebsk',
telegram: '@JanChorny'
}
const first = person.first
const last = person.last
...
							
						
							
After

							
let { first, last } = person
console.log(`${first} ${last}`)// Jan Chorny
let { first: name, last: surname} = person
console.log(name, surname) //Jan Chorny
